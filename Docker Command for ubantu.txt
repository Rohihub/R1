Docker command for ubantu
----------------------------------------

# Update system packages
sudo apt update
sudo apt upgrade -y

# Install prerequisites
sudo apt install apt-transport-https ca-certificates curl software-properties-common -y

# Add Docker's official GPG key
curl -fsSL https://download.docker.com/linux/ubuntu/gpg | sudo gpg --dearmor -o /usr/share/keyrings/docker.gpg

# Add Docker repo
echo "deb [arch=$(dpkg --print-architecture) signed-by=/usr/share/keyrings/docker.gpg] https://download.docker.com/linux/ubuntu $(lsb_release -cs) stable" | sudo tee /etc/apt/sources.list.d/docker.list > /dev/null

# Install Docker
sudo apt update
sudo apt install docker-ce docker-ce-cli containerd.io -y


# Start and enable Docker
sudo systemctl start docker
sudo systemctl enable docker

# Verify installation
docker --version

mkdir ~/html
echo "<h1>Hello World from EC2! Rohidas  </h1>" > ~/html/index.html
echo "<h1>welcome to my webpage </h1>" >> ~/html/index.html

sudo docker run -d -p 80:80 -v ~/html:/usr/share/nginx/html:ro nginx

docker --version	         Check Docker version
sudo systemctl status docker	  Check if Docker is running
docker pull <image>	         Download an image from Docker Hub
docker images	                 List downloaded images
docker run <image>	         Run a container from an image
docker ps	                 List running containers
docker ps -a	                 List all containers (including stopped)
docker stop <container_id>	Stop a running container
docker start <container_id>	Start a stopped container
docker rm <container_id>	Remove a container
docker rmi <image_id>	        Remove an image
docker exec -it <container_id> bash	Access container shell
docker logs <container_id>	View container logs
